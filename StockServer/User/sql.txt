CREATE TABLE `tb_user_fund_detail` (  `userid` bigint(15) NOT NULL COMMENT '用户编号',  `time` datetime NOT NULL COMMENT '充值时间',  `fund` int(11) DEFAULT NULL COMMENT '充值金额',  `grantLevel` int(11) DEFAULT NULL COMMENT '选择的权限等级',  `sequence` varchar(64) DEFAULT NULL COMMENT '序列号或操作者标识',  `channel` int(11) DEFAULT NULL COMMENT '充值渠道',  PRIMARY KEY (`userid`,`time`)) ENGINE=MyISAM DEFAULT CHARSET=utf8;CREATE TABLE `tb_user_grant` (  `grantLevel` int(11) NOT NULL COMMENT '权限等级',  `grantList` text COMMENT '此等级的权限列表',  `fundPerMonth` int(11) DEFAULT NULL COMMENT '权限每月费用',  PRIMARY KEY (`grantLevel`)) ENGINE=MyISAM DEFAULT CHARSET=utf8;CREATE TABLE `tb_user_info` (  `userID` bigint(15) NOT NULL COMMENT '注册时生成的用户编号',  `name` varchar(64) NOT NULL COMMENT '注册时写入的用户名',  `email` varchar(64) NOT NULL COMMENT '注册用邮箱，接收验证码',  `loginTime` datetime DEFAULT NULL COMMENT '最后一次登录时间',  `createTime` datetime DEFAULT NULL COMMENT '注册时间',  `userLevel` int(11) DEFAULT NULL COMMENT '用户等级',  `grantLevel` int(11) DEFAULT NULL COMMENT '权限等级',  `grantTime` date DEFAULT NULL COMMENT '权限到期时间',  `totalFund` int(11) DEFAULT NULL COMMENT '累积充值金额',  `lastFund` int(11) DEFAULT NULL COMMENT '最近充值金额',  `password` varchar(40) NOT NULL,  `fundtime` datetime DEFAULT NULL COMMENT '最近充值日期',  PRIMARY KEY (`userID`),  UNIQUE KEY `name` (`name`),  UNIQUE KEY `email` (`email`)) ENGINE=MyISAM AUTO_INCREMENT=10000 DEFAULT CHARSET=utf8;CREATE TABLE `tb_user_register` (  `name` varchar(64) NOT NULL,  `email` varchar(64) DEFAULT NULL,  `password` varchar(40) DEFAULT NULL,  `verifyCode` varchar(6) DEFAULT NULL,  `time` datetime DEFAULT NULL,  PRIMARY KEY (`name`),  UNIQUE KEY `email` (`email`)) ENGINE=MyISAM DEFAULT CHARSET=utf8;CREATE TABLE `tb_user_session` (  `sessionid` varchar(32) NOT NULL COMMENT 'session编号',  `userid` bigint(15) DEFAULT NULL COMMENT '用户编号',  `time` datetime DEFAULT NULL COMMENT '生成时间',  PRIMARY KEY (`sessionid`)) ENGINE=MyISAM DEFAULT CHARSET=utf8;
